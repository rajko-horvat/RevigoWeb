#!/bin/sh

# Shell script for programatic access to Revigo. Run it with:
# revigo.sh example.csv

# Submit job to Revigo
jobid=$(curl "http://revigo.irb.hr/StartJob.aspx" -X POST --silent --data-urlencode "cutoff=0.7" --data-urlencode "valueType=pvalue" --data-urlencode "speciesTaxon=0" --data-urlencode "measure=SIMREL" --data-urlencode "goList@$1" --header "Content-Type: application/x-www-form-urlencoded" | jq '.jobid')

# Check job status
running=1
while [ $running -ne 0 ]
do
    running=$(curl "http://revigo.irb.hr/QueryJob.aspx" -X POST --silent --data-urlencode "jobid=$jobid" --data-urlencode "type=jstatus" | jq '.running')
    sleep 1
done

# Fetch results
curl "http://revigo.irb.hr/QueryJob.aspx" -X POST --silent --data-urlencode "jobid=$jobid" --data-urlencode "namespace=1" --data-urlencode "type=table"
